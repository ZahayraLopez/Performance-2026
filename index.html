const ADMIN_KEY = 'CAMBIA_ESTA_CLAVE_SEGURA'; const DEFAULT_SHEET_NAME = '📊 Registro Mensual'; // Ajusta estos índices a tu layout real const HEADER_ROW_INDEX = 6; // fila 7 visual const DATA_START_INDEX = 7; // fila 8 visual function doGet(e) { try { const action = (e.parameter.action || 'read').trim(); if (action !== 'read') { return jsonOutput({ ok: false, error: 'Acción GET no válida.' }); } const spreadsheetId = (e.parameter.spreadsheetId || '').trim(); const gid = (e.parameter.gid || '').trim(); const sheetName = (e.parameter.sheet || DEFAULT_SHEET_NAME).trim(); if (!spreadsheetId) { return jsonOutput({ ok: false, error: 'Falta spreadsheetId.' }); } const sheet = getTargetSheet(spreadsheetId, gid, sheetName); if (!sheet) { return jsonOutput({ ok: false, error: 'No se encontró la hoja.' }); } const values = sheet.getDataRange().getDisplayValues(); if (values.length <= DATA_START_INDEX) { return jsonOutput({ ok: true, sheetName: sheet.getName(), gid: String(sheet.getSheetId()), total: 0, data: [] }); } const rows = values.slice(DATA_START_INDEX); const mappedHeaders = [ 'País', 'Área', 'Rol', 'Colaborador', 'Mes', 'Año', 'KPI1', 'Meta1', 'Real1', 'Cumplimiento1', 'KPI2', 'Meta2', 'Real2', 'Cumplimiento2', 'KPI3', 'Meta3', 'Real3', 'Cumplimiento3', 'KPI4', 'Meta4', 'Real4', 'Cumplimiento4', 'KPI5', 'Meta5', 'Real5', 'Cumplimiento5', 'Promedio %', 'Semáforo', 'Observaciones' ]; const data = rows .map((row, idx) => { const obj = { _rowNumber: idx + DATA_START_INDEX + 1 }; mappedHeaders.forEach((key, i) => { obj[key] = row[i] || ''; }); return obj; }) .filter(r => String(r['Colaborador'] || '').trim() !== ''); return jsonOutput({ ok: true, sheetName: sheet.getName(), gid: String(sheet.getSheetId()), total: data.length, data }); } catch (error) { return jsonOutput({ ok: false, error: error.message }); } } function doPost(e) { try { const payload = JSON.parse(e.postData.contents || '{}'); const action = String(payload.action || '').trim(); const spreadsheetId = String(payload.spreadsheetId || '').trim(); const gid = String(payload.gid || '').trim(); const sheetName = String(payload.sheet || DEFAULT_SHEET_NAME).trim(); const adminKey = String(payload.adminKey || '').trim(); const record = payload.record || {}; if (!spreadsheetId) { return jsonOutput({ ok: false, error: 'Falta spreadsheetId.' }); } if (adminKey !== ADMIN_KEY) { return jsonOutput({ ok: false, error: 'Clave de administrador inválida.' }); } const sheet = getTargetSheet(spreadsheetId, gid, sheetName); if (!sheet) { return jsonOutput({ ok: false, error: 'No se encontró la hoja.' }); } if (action === 'addCollaborator') { return addCollaborator(sheet, record); } if (action === 'addGoal') { return addGoal(sheet, record); } if (action === 'updateResult') { return updateResult(sheet, record); } return jsonOutput({ ok: false, error: 'Acción POST no válida.' }); } catch (error) { return jsonOutput({ ok: false, error: error.message }); } } function addCollaborator(sheet, record) { const row = buildBaseRow(record); sheet.appendRow(row); return jsonOutput({ ok: true, message: 'Colaborador agregado correctamente.' }); } function addGoal(sheet, record) { const collaborator = String(record.colaborador || '').trim(); const mes = String(record.mes || '').trim(); const kpi = String(record.kpi || '').trim(); const meta = record.meta ?? ''; if (!collaborator || !mes || !kpi) { return jsonOutput({ ok: false, error: 'Para crear meta se requiere colaborador, mes y KPI.' }); } const values = sheet.getDataRange().getValues(); let updated = false; for (let i = DATA_START_INDEX; i < values.length; i++) { const row = values[i]; const rowCollaborator = String(row[3] || '').trim(); const rowMes = String(row[4] || '').trim(); if (rowCollaborator === collaborator && rowMes === mes) { for (let block = 0; block < 5; block++) { const base = 6 + block * 4; const currentKpi = String(row[base] || '').trim(); if (!currentKpi || currentKpi === kpi) { sheet.getRange(i + 1, base + 1).setValue(kpi); sheet.getRange(i + 1, base + 2).setValue(meta); updated = true; break; } } if (updated) break; } } if (!updated) { return jsonOutput({ ok: false, error: 'No se encontró un registro del colaborador/mes para agregar la meta.' }); } return jsonOutput({ ok: true, message: 'Meta/KPI agregado correctamente.' }); } function updateResult(sheet, record) { const collaborator = String(record.colaborador || '').trim(); const mes = String(record.mes || '').trim(); const kpi = String(record.kpi || '').trim(); const real = record.real ?? ''; const observaciones = String(record.observaciones || '').trim(); if (!collaborator || !mes) { return jsonOutput({ ok: false, error: 'Para actualizar resultado se requiere colaborador y mes.' }); } const values = sheet.getDataRange().getValues(); let updated = false; for (let i = DATA_START_INDEX; i < values.length; i++) { const row = values[i]; const rowCollaborator = String(row[3] || '').trim(); const rowMes = String(row[4] || '').trim(); if (rowCollaborator === collaborator && rowMes === mes) { if (kpi) { for (let block = 0; block < 5; block++) { const base = 6 + block * 4; const currentKpi = String(row[base] || '').trim(); if (currentKpi === kpi) { const metaCell = sheet.getRange(i + 1, base + 2); const realCell = sheet.getRange(i + 1, base + 3); const pctCell = sheet.getRange(i + 1, base + 4); realCell.setValue(real); const metaValue = Number(metaCell.getValue()); const realValue = Number(real); if (!isNaN(metaValue) && metaValue !== 0 && !isNaN(realValue)) { pctCell.setValue(realValue / metaValue); } updated = true; break; } } } if (observaciones) { sheet.getRange(i + 1, 29).setValue(observaciones); updated = true; } if (updated) { recalculateRow(sheet, i + 1); break; } } } if (!updated) { return jsonOutput({ ok: false, error: 'No se encontró el registro para actualizar.' }); } return jsonOutput({ ok: true, message: 'Resultado actualizado correctamente.' }); } function recalculateRow(sheet, rowNumber) { const row = sheet.getRange(rowNumber, 1, 1, 29).getValues()[0]; const percentages = []; for (let block = 0; block < 5; block++) { const pct = Number(row[9 + block * 4]); if (!isNaN(pct)) percentages.push(pct); } const avg = percentages.length ? percentages.reduce((a, b) => a + b, 0) / percentages.length : 0; sheet.getRange(rowNumber, 27).setValue(avg); let semaforo = 'Rojo'; if (avg >= 0.85) semaforo = 'Verde'; else if (avg >= 0.70) semaforo = 'Amarillo'; sheet.getRange(rowNumber, 28).setValue(semaforo); } function buildBaseRow(record) { return [ record.pais || '', record.area || '', record.rol || '', record.colaborador || '', record.mes || '', new Date().getFullYear(), record.kpi || '', record.meta || '', record.real || '', '', '', '', '', '', '', '', '', '', '', '', '', '', '', '', '', '', '', '', record.observaciones || '' ]; } function getTargetSheet(spreadsheetId, gid, sheetName) { const ss = SpreadsheetApp.openById(spreadsheetId); if (gid) { const byGid = ss.getSheets().find(s => String(s.getSheetId()) === gid); if (byGid) return byGid; } const byName = ss.getSheetByName(sheetName); if (byName) return byName; return ss.getSheets()[0]; } function jsonOutput(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
